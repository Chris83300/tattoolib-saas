private function registerUser(array $userData, array $profileData, string $role): User
{
    DB::beginTransaction();
    
    try {
        $user = User::create([
            'name' => $userData['name'],
            'email' => $userData['email'],
            'password' => Hash::make($userData['password']),
            'role' => $role,
            'status' => $role === 'client' ? 'active' : 'pending_verification',
        ]);

        $this->createProfile($user, $profileData, $role);
        
        DB::commit();
        
        Auth::login($user);
        
        return $user;
    } catch (\Exception $e) {
        DB::rollBack();
        Log::error('Registration failed', ['role' => $role, 'error' => $e->getMessage()]);
        throw $e;
    }
}

public function authorize(): bool
{
    $bookingRequest = $this->route('booking_request');
    $user = auth()->user();
    
    return $user && (
        ($user->isClient() && $bookingRequest->client_id === $user->client->id) ||
        ($user->isTattooer() && $bookingRequest->bookable->user_id === $user->id)
    );
